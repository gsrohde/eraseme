16 // Determine assimilation, stomatal conductance, and Ci following `c3photoC()`.
17 // Here, rather than using the FvCB model for C3 photosynthesis, gross
18 // assimilation is determined using a simple RUE model. Water stress is ignored.
19 // Respiration and stomatal conductance are otherwise calculated using the same
20 // methods as `c3photoC()`.
21 struct rue_str rue_photo(
22  double Qp, // mol / m^2 / s
23  double alpha_rue, // dimensionless
24  double Tleaf, // degrees C
25  double RH, // dimensionless from Pa / Pa
26  double Rd0, // mol / m^2 / s
27  double bb0, // mol / m^2 / s
28  double bb1, // dimensionless
29  double Ca // dimensionless from mol / mol
30 )
31 {
32  using conversion_constants::celsius_to_kelvin; // deg. C or K
33 
34  // Convert leaf temperature to Kelvin
35  double const tlk = Tleaf + celsius_to_kelvin; // K
36 
37  // Calculate gross assimilation based on incident light and radiation use
38  // efficiency
39  double const ag = Qp * alpha_rue; // mol / m^2 / s
40 
41  // Determine respiration from the leaf temperature using Arrhenius exponents
42  // from Bernacchi et al. (2001) Plant, Cell and Environment, 24(2), 253-259.
43  // https://doi.org/10.1111/j.1365-3040.2001.00668.x
44  double const rd = Rd0 *
45  arrhenius_exponent(
46  18.72,
47  46.39e3,
48  tlk); // mol / m^2 / s
49 
50  // Determine net assimilation
51  double const an = ag - rd; // mol / m^2 / s
52 
53  // Determine stomatal conductance
54  double const gs = ball_berry(an, Ca, RH, bb0, bb1) * 1e-3; // mol / m^2 / s
55 
56  // Determine intercellular CO2 concentration
57  double const ci = Ca - an * 1.6 / gs; // dimensionless
58

62  result.Gs = gs * 1e3; // mmol / m^2 / s
