133 void senescence_logistic::do_operation() const
134 {
135  double senescence_leaf = kSeneLeaf * Leaf; // Mg / ha, amount of leaf senesced
136  double senescence_stem = kSeneStem * Stem; // Mg / ha, amount of stem senesced
137  double senescence_root = kSeneRoot * Root; // Mg / ha, amount of root senesced
138  double senescence_rhizome = kSeneRhizome * Rhizome; // Mg / ha, amount of
139  // rhizome senesced
140 
141  // change in leaf biomass = minus amount senesced + new leaf tissue from
142  // remobilized amount (Allows for leaves to start senescing while new leaves
143  // are still being produced).
144  double dLeaf = -senescence_leaf + kLeaf * senescence_leaf * remobilization_fraction; // Mg / ha
145 
146  // change in amount of leaf litter
147  double dLeafLitter = senescence_leaf * (1 - remobilization_fraction); // Mg / ha
148 
149  // change in stem biomass = minus amount senesced + new stem tissue from
150  // remobilized leaf fraction
151  double dStem = -senescence_stem + kStem * senescence_leaf * remobilization_fraction; // Mg / ha
152 
153  double dStemLitter = senescence_stem; // Mg / ha, change in amount of stem litter
154 
155  // change in root biomass = minus amount senesced + new root tissue from remobilized
156  // leaf fraction
157  double dRoot = -senescence_root + kRoot * senescence_leaf * remobilization_fraction; // Mg / ha
158 
159  double dRootLitter = senescence_root; // Mg / ha, change in amount of root litter
160 
161  // change in rhizome biomass = minus amount senesced + new rhizome from remobilized
162  // leaf fraction
163  double dRhizome = -senescence_rhizome + kRhizome * senescence_leaf * remobilization_fraction; // Mg / ha
164  double dRhizomeLitter = senescence_rhizome; // Mg / ha, change in rhizome litter
165 
166  // change in grain biomass = new grain from remobilized leaf fraction.
167  // currently do not include grain senescence.
168  double dGrain = kGrain * senescence_leaf * remobilization_fraction; // Mg / ha
