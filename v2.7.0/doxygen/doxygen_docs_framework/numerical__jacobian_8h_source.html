84  // Perturb each element x_i of the input vector to find df_j(x,t)/dx_i,
85  // which is stored at jacobi(j,i)
86  double h;
87  x_vector_type x_perturbed = x;
88 
89  for (size_t i = 0; i < n; i++) {
90  // Detemine the step size h by taking a fraction of x[i]: h = x[i] * eps_deriv.
91  // Ensure that the step size h is close to this value but is exactly representable
92  // (see Numerical Recipes in C, 2nd ed., Section 5.7)
93  h = x[i] * calculation_constants::eps_deriv;
94  if (h == 0.0) {
95  h = calculation_constants::eps_deriv * calculation_constants::eps_deriv;
96  }
97  double temp = x[i] + h;
98  h = temp - x[i];
99 
100  // Calculate the new function value
101  x_perturbed[i] = x[i] + h; // Add h to the ith differential quantity
102  evaluate_equations(equation_ptr, x_perturbed, t, f_perturbed); // Calculate f_perturbed
103 
104  // Store the results in the Jacobian matrix
105  for (size_t j = 0; j < n; j++) {
106  jacobi(j, i) = (f_perturbed[j] - f_current[j]) / h;
107  }

109  // Reset the ith differential quantity
