94 void multilayer_canopy_properties::run() const
95 {
96  // Calculate values of incident photosynthetically active photon flux
97  // density (PPFD) and absorbed shortwave energy throughout the canopy. Note
98  // that the `sunML` function expects input expects PPFD values, so we must
99  // convert photosynthetically active radiation (PAR) to PPFD using the
100  // energy content of light in the PAR band
101  struct Light_profile light_profile = sunML(
102  par_incident_direct / par_energy_content, // micromol / (m^2 beam) / s
103  par_incident_diffuse / par_energy_content, // micromol / m^2 / s
104  lai,
105  nlayers,
106  cosine_zenith_angle,
107  kd,
108  chil,
109  absorptivity_par,
110  heightf,
111  par_energy_content,
112  par_energy_fraction,
113  leaf_transmittance,
114  leaf_reflectance);
115 
116  // Calculate relative humidity levels throughout the canopy
117  double relative_humidity_profile[nlayers];
118  RHprof(rh, nlayers, relative_humidity_profile); // Modifies relative_humidity_profile
119 
120  // Calculate windspeed throughout the canopy
121  double wind_speed_profile[nlayers];
122  WINDprof(windspeed, lai, nlayers, wind_speed_profile); // Modifies wind_speed_profile
123 
124  // Calculate leaf nitrogen throughout the canopy
125  double leafN_profile[nlayers];
126  LNprof(LeafN, lai, nlayers, kpLN, leafN_profile); // Modifies leafN_profile
127 
128  // Don't calculate anything based on the nitrogen profile
129  if (lnfun != 0) {
130  throw std::logic_error("Thrown by the multilayer_canopy_properties module: lnfun != 0 is not yet supported.");
131  }
132 
133  // Update layer-dependent outputs
134  for (int i = 0; i < nlayers; ++i) {
135  update(sunlit_fraction_ops[i], light_profile.sunlit_fraction[i]);
136  update(sunlit_incident_ppfd_ops[i], light_profile.sunlit_incident_ppfd[i]);
137  update(sunlit_absorbed_shortwave_ops[i], light_profile.sunlit_absorbed_shortwave[i]);
138 
139  update(shaded_fraction_ops[i], light_profile.shaded_fraction[i]);
140  update(shaded_incident_ppfd_ops[i], light_profile.shaded_incident_ppfd[i]);
141  update(shaded_absorbed_shortwave_ops[i], light_profile.shaded_absorbed_shortwave[i]);
142 
143  update(average_incident_ppfd_ops[i], light_profile.average_incident_ppfd[i]);
144  update(average_absorbed_shortwave_ops[i], light_profile.average_absorbed_shortwave[i]);

147  update(height_ops[i], light_profile.height[i]);
148  update(rh_ops[i], relative_humidity_profile[i]);
