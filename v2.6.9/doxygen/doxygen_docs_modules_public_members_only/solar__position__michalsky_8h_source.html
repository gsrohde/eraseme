184 void solar_position_michalsky::do_operation() const
185 {
186  // Define some constants
187  double constexpr deg_to_hr = 1.0 / 15.0;
188  double constexpr hr_to_deg = 15.0;
189  double constexpr hr_per_day = 24.0;
190  double constexpr deg_per_rev = 360.0;
191 
192  double constexpr jd_ref_1948 = 2432916.5; // Julian date at midnight on 31 December 1948 (UTC)
193  double constexpr jd_ref_2000 = 2451545.0; // Julian date at noon on 1 January 2000 (UTC)
194 
195  // Unpack the doy and hour in UTC
196  double time_utc = time - time_zone_offset / hr_per_day; // days
197  double const doy_utc = floor(time_utc); // days
198  double const hour_utc = hr_per_day * (time_utc - doy_utc); // hours
199 
200  // Calculate the Julian date
201  double const delta = year - 1949.0;
202  double const leap = floor(0.25 * delta);
203  double const jd = jd_ref_1948 + delta * 365.0 + leap + doy_utc + hour_utc / hr_per_day; // days
204 
205  // Calculate the "n" date
206  double const n = jd - jd_ref_2000; // days
207 
208  // Calculate the ecliptic coordinates of the sun
209  double const L = fmod(280.460 + 0.9856474 * n, deg_per_rev); // degrees
210  double const g = fmod(357.528 + 0.9856003 * n, deg_per_rev); // degrees
211  double const ell = fmod(L + 1.915 * sin_deg(g) + 0.020 * sin_deg(2 * g), deg_per_rev); // degrees
212  double const ep = 23.439 - 0.0000004 * n; // degrees
213 
214  // Calculate the equatorial celestial coordinates of the sun
215  double const ra = atan2_deg(cos_deg(ep) * sin_deg(ell), cos_deg(ell)); // degrees
216  double const dec = asin_deg(sin_deg(ep) * sin_deg(ell)); // degrees
217 
218  // Calculate the sidereal time
219  double const gmst = fmod(6.697375 + 0.0657098242 * n + hour_utc, hr_per_day); // hours
220  double const lmst = fmod(gmst + longitude * deg_to_hr, hr_per_day); // hours
221 
222  // Convert to local coordinates
223  double const lha = fmod(lmst * hr_to_deg - ra, deg_per_rev); // degrees
224 
225  double const zen = acos_deg(sin_deg(dec) * sin_deg(lat) +
226  cos_deg(dec) * cos_deg(lat) * cos_deg(lha)); // degrees
227 
228  double az = asin_deg(-cos_deg(dec) * sin_deg(lha) / cos_deg(90.0 - zen)); // degrees
229 
230  // Make sure azimuth is in the correct quadrant
231  double const el = 90.0 - zen; // degrees
232  double const el_critical = asin_deg(sin_deg(dec) / sin_deg(lat)); // degrees
233 
234  if (el >= el_critical) {
235  az = 180.0 - az; // degrees
236  } else if (lha > 0) {
237  az = 360.0 + az; // degrees
238  }
239 
240  // Determine the cosine of the zenith angle
241  double const cos_zen = cos_deg(zen); // dimensionless
242 
243  // Update the output pointers
244  update(cosine_zenith_angle_op, cos_zen);
245  update(julian_date_op, jd);
246  update(solar_L_op, L);
247  update(solar_g_op, g);
248  update(solar_ell_op, ell);
249  update(solar_ep_op, ep);
250  update(solar_ra_op, ra);
251  update(solar_dec_op, dec);
252  update(gmst_op, gmst);
253  update(lmst_op, lmst);
254  update(lha_op, lha);
255  update(solar_zenith_angle_op, zen);
256  update(solar_azimuth_angle_op, az);
257 }
