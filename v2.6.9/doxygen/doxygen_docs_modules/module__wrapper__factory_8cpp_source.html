196 string_vector module_wrapper_factory::get_modules()
197 {
198  string_vector module_name_vector;
199  for (auto const& x : module_wrapper_creators) {
200  module_name_vector.push_back(x.first);
201  }
202 
203  auto case_insensitive_compare = [](std::string const& a, std::string const& b) {
204  // Make a lowercase copy of a
205  std::string al = a;
206  std::transform(al.begin(), al.end(), al.begin(), [](unsigned char c) { return std::tolower(c); });
207 
208  // Make a lowercase copy of b
209  std::string bl = b;
210  std::transform(bl.begin(), bl.end(), bl.begin(), [](unsigned char c) { return std::tolower(c); });
211 
212  int compare = al.compare(bl);
213  return (compare > 0) ? false : true;
214  };
215 
216  std::sort(module_name_vector.begin(), module_name_vector.end(), case_insensitive_compare);
217 
218  return module_name_vector;
219 }

221 std::unordered_map<std::string, string_vector> module_wrapper_factory::get_all_quantities()
222 {
223  // Make the output map
224  std::unordered_map<std::string, string_vector> quantity_map = {
225  {"module_name", string_vector{}},
226  {"quantity_type", string_vector{}},
227  {"quantity_name", string_vector{}}
228  };
229 
230  // Make a lambda function for adding entries to the map
231  auto add_quantity_map_entry = [&quantity_map](std::string module_name, std::string quantity_type, std::string quantity_name) {
232  quantity_map["module_name"].push_back(module_name);
233  quantity_map["quantity_type"].push_back(quantity_type);
234  quantity_map["quantity_name"].push_back(quantity_name);
235  };
236 
237  // Fill the output map with all the quantities
238  for (std::string const& module_name : module_wrapper_factory::get_modules()) {
239  auto w = module_wrapper_factory::create(module_name);
240 
241  // Add the module's inputs to the parameter map
242  for (std::string const& input_name : w->get_inputs()) {
243  add_quantity_map_entry(module_name, "input", input_name);
244  }
245 
246  // Add the module's outputs to the parameter map
247  for (std::string const& output_name : w->get_outputs()) {
248  add_quantity_map_entry(module_name, "output", output_name);
249  }
250  }
251 
252  return quantity_map;
253 }
