123 void oscillator_clock_calculator::do_operation() const
124 {
126  // Collect inputs and make calculations //
128 
129  using math_constants::pi;
130 
131  // Get the current time value
132  const double time = *time_ip;
133  const double hour = 24.0 * (time - floor(time));
134 
135  // Get the current light value
136  const double light = *light_ip;
137 
138  // Get the current state of the night and day trackers
139  const double night_tracker = *night_tracker_ip;
140  const double day_tracker = *day_tracker_ip;
141 
142  // Get the kick strength
143  const double kick_strength = *kick_strength_ip;
144 
145  // Calculate the kicks
146  // The dawn kick is created by taking the night tracker value during
147  // the day, which is just a short decay portion
148  // The dusk kick is created by taking the day tracker value during
149  // the night, which is just a short decay portion
150  const double dawn_kick = light * kick_strength * night_tracker;
151  const double dusk_kick = (1.0 - light) * kick_strength * day_tracker;
152 
153  // Get the current state of the dawn and dusk tracking oscillators
154  const double dawn_b = *dawn_b_ip;
155  const double dawn_a = *dawn_a_ip;
156  const double dusk_b = *dusk_b_ip;
157  const double dusk_a = *dusk_a_ip;
158  const double ref_b = *ref_b_ip;
159  const double ref_a = *ref_a_ip;
160 
161  // Calculate the dawn phase angle, which is zero around dawn and increases
162  // throughout the day.
163  const double dawn_phase = range_adjusted_atan2(dawn_b, dawn_a);
164 
165  // Calculate the dusk phase angle, which is zero around dusk and increases
166  // throughout the night.
167  const double dusk_phase = range_adjusted_atan2(dusk_b, dusk_a);
168 
169  // Calculate the reference phase angle, which is not coupled to the light.
170  const double ref_phase = range_adjusted_atan2(ref_b, ref_a);
171 
172  // Calculate the day and night length indicators (in hours):
173  const double day_length = dusk_phase > dawn_phase ? (dawn_phase - dusk_phase + 2 * pi) * 12.0 / pi
174  : (dawn_phase - dusk_phase) * 12.0 / pi;
175  const double night_length = dawn_phase > dusk_phase ? (dusk_phase - dawn_phase + 2 * pi) * 12.0 / pi
176  : (dusk_phase - dawn_phase) * 12.0 / pi;
177 
178  // Calculate the sunrise and sunset times
179  const double sunrise = dawn_phase * 12 / pi < hour ? (hour - dawn_phase * 12 / pi)
180  : (hour - dawn_phase * 12 / pi + 24.0);
181  const double sunset = dusk_phase * 12 / pi < hour ? (hour - dusk_phase * 12 / pi)
182  : (hour - dusk_phase * 12 / pi + 24.0);
183 
185  // Update the output quantity list //
187 
188  update(dawn_kick_op, dawn_kick);
189  update(dusk_kick_op, dusk_kick);
190  update(dawn_phase_op, dawn_phase);
191  update(dusk_phase_op, dusk_phase);
192  update(ref_phase_op, ref_phase);
193  update(dawn_radius_op, sqrt(dawn_a * dawn_a + dawn_b * dawn_b));
194  update(dusk_radius_op, sqrt(dusk_a * dusk_a + dusk_b * dusk_b));
195  update(ref_radius_op, sqrt(ref_a * ref_a + ref_b * ref_b));
196  update(day_length_op, day_length);
197  update(night_length_op, night_length);
198  update(sunrise_op, sunrise);
199  update(sunset_op, sunset);
200 }
