79 void shortwave_atmospheric_scattering::do_operation() const
80 {
81  // Calculate diffuse and direct irradiance transmittance
82  double irradiance_direct_transmittance;
83  double irradiance_diffuse_transmittance;
84  if (cosine_zenith_angle <= 0) {
85  // In this case, the sun is at or below the horizon, so there is no direct irradiance
86  irradiance_direct_transmittance = 0.0;
87  irradiance_diffuse_transmittance = 1.0;
88  } else {
89  // The sun is above the horizon, so use its zenith angle to calculate transmittance
90  // The direct transmission is calculated using Equation 11.11 from Campbell & Norman (1998), solving for
91  // irradiance_direct_transmittance = S_p / S_p0
92  // The indirect (AKA diffuse) transmission is calculated using Equation 11.13 from Campbell & Norman (1998), solving for
93  // irradiance_diffuse_transmittance = S_d / S_p0
94  const double pressure_ratio = atmospheric_pressure / physical_constants::atmospheric_pressure_at_sea_level; // dimensionless.
95  irradiance_direct_transmittance = pow(atmospheric_transmittance, (pressure_ratio / cosine_zenith_angle)); // Modified from equation 11.11 of Norman and Campbell.
96  irradiance_diffuse_transmittance = atmospheric_scattering * (1.0 - irradiance_direct_transmittance) * cosine_zenith_angle; // Modified from equation 11.13 of Norman and Campbell.
97  }
98 
99  // Calculate the diffuse and direct fractions
100  const double irradiance_direct_fraction = irradiance_direct_transmittance / (irradiance_direct_transmittance + irradiance_diffuse_transmittance);
101  const double irradiance_diffuse_fraction = 1.0 - irradiance_direct_fraction;
102 
103  // Update the outputs
104  update(irradiance_direct_transmittance_op, irradiance_direct_transmittance);
105  update(irradiance_diffuse_transmittance_op, irradiance_diffuse_transmittance);
106  update(irradiance_direct_fraction_op, irradiance_direct_fraction);
107  update(irradiance_diffuse_fraction_op, irradiance_diffuse_fraction);
108 }
