89 void one_layer_soil_profile_derivatives::do_operation() const
90 {
91  double soil_water_potential = -exp(log(0.033) + ((log(soil_field_capacity) - log(soil_water_content)) / (log(soil_field_capacity) - log(soil_wilting_point)) * (log(1.5) - log(0.033)))) * 1e3; // kPa.
92 
93  double hydraulic_conductivity = soil_saturated_conductivity * pow(soil_air_entry / soil_water_potential, 2 + 3 / soil_b_coefficient); // Kg s m^-3.
94  double J_w = -hydraulic_conductivity * (-soil_water_potential / (soil_depth * 0.5)) - acceleration_from_gravity * hydraulic_conductivity; // kg m^-2 s^-1. Campbell, pg 129. I multiply soil depth by 0.5 to calculate the average depth.
95 
96  double density_of_water_at_20_celcius = 998.2; // kg m^-3
97 
98  double drainage = J_w / density_of_water_at_20_celcius; // m s^-1.
99  double precipitation_m_s = precipitation_rate * 1e-3; // m s^-2.
100 
101  double second_per_hour = 3600;
102  double runoff = (soil_water_content - soil_saturation_capacity) * soil_depth / second_per_hour; // m s^-1. The previous model drained everything in a single time-step. By default the time step was one hour, so use a rate that drains everything in one hour.
103  double n_leach = runoff / (1e3 * physical_constants::molar_mass_of_water) * (0.2 + 0.7 * soil_sand_content) / second_per_hour; // Base the rate on an hour for the same reason as was used with `runoff`.
104 
105  double evapotranspiration_volume = evapotranspiration / density_of_water_at_20_celcius / 1e4 / second_per_hour; // m^3 m^-2 s^-1
106 
107  // Update the output quantity list
108  update(soil_water_content_op, (drainage + precipitation_m_s - runoff - evapotranspiration_volume) / soil_depth);
109  update(soil_n_content_op, -n_leach);
110 }
