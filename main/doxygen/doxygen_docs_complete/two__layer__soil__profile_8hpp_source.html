155 void two_layer_soil_profile::do_operation() const
156 {
157  // Collect inputs and make calculations
158  double cws[] = {cws1, cws2};
159  double soil_depths[] = {soil_depth1, soil_depth2, soil_depth3};
160 
161  struct soilML_str soilMLS = soilML(
162  precip, canopy_transpiration_rate, cws, soil_depth3, soil_depths,
163  soil_field_capacity, soil_wilting_point, soil_saturation_capacity,
164  soil_air_entry, soil_saturated_conductivity, soil_b_coefficient,
165  soil_sand_content, phi1, phi2, wsFun, 2 /* Always uses 2 layers */,
166  Root, lai, 0.68, temp, solar, windspeed, rh, hydrDist, rfl, rsec, rsdf,
167  soil_clod_size, soil_reflectance, soil_transmission,
168  specific_heat_of_air, par_energy_content);
169 
170  double layer_one_depth = soil_depth2 - soil_depth1;
171  double layer_two_depth = soil_depth3 - soil_depth2;
172 
173  double cws_mean =
174  (soilMLS.cws[0] * layer_one_depth + soilMLS.cws[1] * layer_two_depth) /
175  (layer_one_depth + layer_two_depth);
176 
177  // Update the output quantity list
178  update(cws1_op, soilMLS.cws[0] - cws1);
179  update(cws2_op, soilMLS.cws[1] - cws2);
180  update(soil_water_content_op, cws_mean - soil_water_content);
181 }
