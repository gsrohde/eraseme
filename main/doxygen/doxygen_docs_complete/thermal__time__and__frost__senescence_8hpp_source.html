250 void thermal_time_and_frost_senescence::do_operation() const
251 {
252  // Add the new tissue growth to the history vectors
253  assim_rate_stem_vec.push_back(net_assimilation_rate_stem);
254  assim_rate_root_vec.push_back(net_assimilation_rate_root);
255  assim_rate_rhizome_vec.push_back(net_assimilation_rate_rhizome);
256 
257  // Initialize variables
258  double dLeafdeathrate{0.0};
259  double dLeaf{0.0};
260  double dStem{0.0};
261  double dRoot{0.0};
262  double dRhizome{0.0};
263  double dGrain{0.0};
264  double dLeafLitter{0.0};
265  double dStemLitter{0.0};
266  double dRootLitter{0.0};
267  double dRhizomeLitter{0.0};
268  double dstem_senescence_index{0.0};
269  double droot_senescence_index{0.0};
270  double drhizome_senescence_index{0.0};
271 
272  // Calculate the leaf death rate due to frost
273  double frost_leaf_death_rate{0.0};
274  if (TTc >= seneLeaf) { // Leaf senescence has started
275  bool A = lat >= 0.0; // In Northern hemisphere
276  bool B = time >= 180.0; // In second half of the year
277  if ((A && B) || ((!A) && (!B))) { // Winter in either hemisphere
278  // frost_leaf_death_rate changes linearly from 100 to 0 as temp
279  // changes from Tfrostlow to Tfrosthigh and is limited to [0,100]
280  std::max(0.0, std::min(100.0, 100.0 * (Tfrosthigh - temp) /
281  (Tfrosthigh - Tfrostlow)));
282  }
283  }
284 
285  // The current leaf death rate is the larger of the previously stored leaf
286  // death rate and the new frost death rate. I.e., the leaf death rate
287  // sometimes increases but never decreases
288  double current_leaf_death_rate = std::max(leafdeathrate,
289  frost_leaf_death_rate);
290 
291  // Report the change in leaf death rate as the derivative of leafdeathrate.
292  // Note: this will probably only work well with the Euler method
293  dLeafdeathrate = current_leaf_death_rate - leafdeathrate;
294 
295  // Calculate leaf senescence
296  if (TTc >= seneLeaf) {
297  // Use the leaf death rate to determine the change in leaf mass
298  double change = Leaf * current_leaf_death_rate * (0.01 / 24);
299 
300  // Remobilize some of the lost leaf tissue and send the rest to the
301  // litter. EBL wonders: why does the leaf remobilize its own tissue?
302  dLeaf += -change + kLeaf * change * remobilization_fraction;
303  dLeafLitter += change * (1.0 - remobilization_fraction);
304  dRhizome += kRhizome * change * remobilization_fraction;
305  dStem += kStem * change * remobilization_fraction;
306  dRoot += kRoot * change * remobilization_fraction;
307  dGrain += kGrain * change * remobilization_fraction;
308  }
309 
310  // Calculate stem senescence
311  if (TTc >= seneStem) {
312  // Look back in time to find out how much the tissue grew in the past.
313  double change = assim_rate_stem_vec.at(stem_senescence_index);
314 
315  // Subtract the new growth from the tissue derivative
316  dStem -= change;
317 
318  // Send the lost tissue to the litter
319  dStemLitter += change;
320 
321  // Increment the tissue senescence index
322  dstem_senescence_index++;
323  }
324 
325  // Calculate root senescence
326  if (TTc >= seneRoot) {
327  // Look back in time to find out how much the tissue grew in the past.
328  double change = assim_rate_root_vec.at(root_senescence_index);
329 
330  // Subtract the new growth from the tissue derivative
331  dRoot -= change;
332 
333  // Send the lost tissue to the litter
334  dRootLitter += change;
335 
336  // Increment the tissue senescence index
337  droot_senescence_index++;
338  }
339 
340  if (kRhizome < 0) {
341  // Increment the rhizome senescence index if it is acting as a source
342  drhizome_senescence_index++;
343  }
344 
345  // Calculate rhizome senescence
346  if (TTc >= seneRhizome) {
347  // Look back in time to find out how much the tissue grew in the past.
348  double change = assim_rate_rhizome_vec.at(rhizome_senescence_index);
349 
350  // Subtract the new growth from the tissue derivative
351  dRhizome -= change;
352 
353  // Send the lost tissue to the litter
354  dRhizomeLitter += change;
355 
356  // Increment the tissue senescence index
357  drhizome_senescence_index++;
358  }
359 
360  // Update the output quantity list
361  update(leafdeathrate_op, dLeafdeathrate);
362 
363  update(Leaf_op, dLeaf);
364  update(Stem_op, dStem);
