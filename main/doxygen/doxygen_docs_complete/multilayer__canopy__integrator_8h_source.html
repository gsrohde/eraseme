142 void multilayer_canopy_integrator::run() const
143 {
144  double const LAIc = lai / nlayers;
145  double canopy_assimilation_rate = 0;
146  double canopy_transpiration_rate = 0;
147  double canopy_conductance = 0;
148  double GrossAssim = 0;
149 
150  // Integrate assimilation, transpiration, and conductance throughout the
151  // canopy
152  for (int i = 0; i < nlayers; ++i) {
153  double const sunlit_lai = *sunlit_fraction_ips[i] * LAIc;
154  double const shaded_lai = *shaded_fraction_ips[i] * LAIc;
155 
156  canopy_assimilation_rate += *sunlit_Assim_ips[i] * sunlit_lai +
157  *shaded_Assim_ips[i] * shaded_lai;
158 
159  canopy_transpiration_rate += *sunlit_TransR_ips[i] * sunlit_lai +
160  *shaded_TransR_ips[i] * shaded_lai;
161 
162  canopy_conductance += *sunlit_Gs_ips[i] * sunlit_lai +
163  *shaded_Gs_ips[i] * shaded_lai;
164 
165  GrossAssim += *sunlit_GrossAssim_ips[i] * sunlit_lai +
166  *shaded_GrossAssim_ips[i] * shaded_lai;
167  }
168 
169  // Modify net assimilation to account for respiration
170  // Note: this was originally only done for the C3 canopy
171  // Note: it seems like this should not be necessary since the assimilation
172  // model includes respiration
173  canopy_assimilation_rate *= (1.0 - growth_respiration_fraction);
174 
175  // For assimilation, we need to convert micromol / m^2 / s into
176  // Mg / ha / hr, assuming that all carbon is converted into biomass in the
177  // form of glucose (C6H12O6), i.e., six assimilated CO2 molecules contribute
178  // one glucose molecule. Using the molar mass of glucose in kg / mol, the
179  // conversion can be accomplished with the following factor:
180  // (1 glucose / 6 CO2) * (3600 s / hr) * (1e-6 mol / micromol) *
181  // (1e-3 Mg / kg) * (1e4 m^2 / ha)
182  // = 6e-3 s * mol * Mg * m^2 / (hr * micromol * kg * ha)
183  const double cf = physical_constants::molar_mass_of_glucose * 6e-3; // (Mg / ha / hr) / (micromol / m^2 / s)
184 
185  // For transpiration, we need to convert mmol / m^2 / s into Mg / ha / hr
186  // using the molar mass of water in kg / mol, which can be accomplished by
187  // the following conversion factor:
188  // (3600 s / hr) * (1e-3 mol / mmol) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
189  // = 36 s * mol * Mg * m^2 / (hr * mmol * kg * ha)
190  const double cf2 = physical_constants::molar_mass_of_water * 36; // (Mg / ha / hr) / (mmol / m^2 / s)
191 
192  update(canopy_assimilation_rate_op, canopy_assimilation_rate * cf);
193 
194  update(GrossAssim_op, GrossAssim * cf);
195 
196  update(canopy_transpiration_rate_op, canopy_transpiration_rate * cf2);
197 
198  update(canopy_conductance_op, canopy_conductance);
199 }
