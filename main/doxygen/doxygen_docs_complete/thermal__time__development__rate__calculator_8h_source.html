108 {
109  // Calculate the development_rate
110  double development_rate; // day^-1
111  double temp_diff = temp - tbase; // degrees C
112  temp_diff = (temp_diff > 0) ? temp_diff : 0; // if temp < tbase, temp_diff = 0
113 
114  if (time < sowing_time) {
115  // The seeds haven't been sown yet, so no development should occur
116  development_rate = 0; // day^-1
117  } else if (DVI < -1) {
118  // error, DVI out of bounds, this should never occur unless initial DVI
119  // state is less than -1.
120  development_rate = 0;
121  } else if (DVI < 0) {
122  // 1. Sowing to emergence
123  development_rate = temp_diff / TTemr; // day^-1
124 
125  } else if (DVI < 1) {
126  // 2. Vegetative stages
127  development_rate = temp_diff / TTveg; // day^-1
128 
129  } else {
