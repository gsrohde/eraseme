235  assim_rate_stem_vec.push_back(net_assimilation_rate_stem);

253 
254  if (TTc >= seneLeaf) {
255  // Look back in time to find out how much the tissue grew in the past
256  double change = assim_rate_leaf_vec.at(leaf_senescence_index);
257 
258  // Subtract the rate of new growth that occurred in the past from the
259  // derivative
260  dLeaf -= change;
261 
262  // Remobilize some of the lost tissue and send the rest to the litter
263  dLeafLitter += change * (1.0 - remobilization_fraction);
264  dRhizome += kRhizome * change * remobilization_fraction;
265  dStem += kStem * change * remobilization_fraction;
266  dRoot += kRoot * change * remobilization_fraction;
267  dGrain += kGrain * change * remobilization_fraction;
268 
269  // Increment the tissue senescence index
270  dleaf_senescence_index++;
271  }
272 
273  if (TTc >= seneStem) {

275  double change = assim_rate_stem_vec.at(stem_senescence_index);
