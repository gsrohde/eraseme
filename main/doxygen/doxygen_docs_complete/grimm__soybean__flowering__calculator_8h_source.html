151 void grimm_soybean_flowering_calculator::do_operation() const
152 {
153  // Calculate the night length
154  const double night_length = 24.0 - day_length; // hours
155 
156  // Calculate the normalized growth rate based on the day of year, night
157  // length, temperature, and growth stage
158  double rate_normalized = 0.0; // dimensionless
159  double temperature_rate_normalized = 0.0; // dimensionless
160  double photoperiod_rate_normalized = 0.0; // dimensionless
161  if (time < sowing_time) {
162  // The plant has not been sowed yet, so it can't grow
163  rate_normalized = 0.0;
164  } else if (grimm_physiological_age < grimm_juvenile_pd_threshold) {
165  // The plant is still in the juvenile (photoperiod-insensitive) phase
166  if (temp < grimm_juvenile_T0) {
167  // The temperature is too low for the plant to grow
168  rate_normalized = 0.0;
169  } else if (temp < grimm_juvenile_T1) {
170  // There is a linear increase from 0 to 1 as temp changes from T0 to
171  // T1
172  rate_normalized = (temp - grimm_juvenile_T0) /
173  (grimm_juvenile_T1 - grimm_juvenile_T0);
174  } else if (temp < grimm_juvenile_T2) {
175  // We are in the flat region at maximum growth
176  rate_normalized = 1.0;
177  } else if (temp < grimm_juvenile_T3) {
178  // There is a linear decrease from 1 to 0 as temp changes from T2 to
179  // T3
180  rate_normalized = 1.0 - (temp - grimm_juvenile_T2) /
181  (grimm_juvenile_T3 - grimm_juvenile_T2);
182  } else {
183  // The temperature is too high for the plant to grow
184  rate_normalized = 0.0;
185  }
186  } else {
187  // The plant has entered the photoperiod-sensitive phase
188  if (temp < grimm_T_min) {
189  // The temperature is too low for the plant to develop
190  temperature_rate_normalized = 0.0;
191  } else if (temp < grimm_T_opt) {
192  // There is a linear increase from 0 to 1 as temp changes from T_min
193  // to T_opt
194  temperature_rate_normalized = (temp - grimm_T_min) /
195  (grimm_T_opt - grimm_T_min);
196  } else {
197  // The development rate is maxed out at 1
198  temperature_rate_normalized = 1.0;
199  }
200  if (night_length < grimm_N_min) {
201  // The night is too short for the plant to develop
202  photoperiod_rate_normalized = 0.0;
203  } else if (night_length < grimm_N_opt) {
204  // There is a linear increase from 0 to 1 as night_length changes
205  // from N_min to N_opt
206  photoperiod_rate_normalized = (night_length - grimm_N_min) /
207  (grimm_N_opt - grimm_N_min);
208  } else {
209  // The development rate is maxed out at 1
210  photoperiod_rate_normalized = 1.0;
211  }
212  rate_normalized = temperature_rate_normalized * photoperiod_rate_normalized;
213  }
214 
215  // Decide if the plant is flowering
216  const bool is_flowering =
217  (grimm_physiological_age > grimm_juvenile_pd_threshold + grimm_flowering_threshold);
218 
219  // Scale the rates to the maximum value, which is 1 physiological day per
220  // day, equivalent to 1 / 24 physiological days per hour
221  const double rate_per_hour = rate_normalized / 24.0;
222  const double temperature_rate_per_hour = temperature_rate_normalized / 24.0;
223  const double photoperiod_rate_per_hour = photoperiod_rate_normalized / 24.0;
224 
225  // Update the output quantity list
226  update(grimm_rate_op, rate_per_hour);
227  update(grimm_rate_temperature_op, temperature_rate_per_hour);
228  update(grimm_rate_photoperiod_op, photoperiod_rate_per_hour);
229  update(grimm_flowering_op, is_flowering);
230 }
