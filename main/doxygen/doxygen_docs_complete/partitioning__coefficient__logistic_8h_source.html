124 void partitioning_coefficient_logistic::do_operation() const
125 {
126  // Determine partitioning coefficients using multinomial logistic equations
127  // from Osborne et al., 2015 JULES-crop https://doi.org/10.5194/gmd-8-1139-2015
128 
129  // denominator term for kRoot, kStem, kLeaf, and kGrain
130  double kDenom = exp(alphaRoot + betaRoot * DVI) + exp(alphaLeaf + betaLeaf * DVI) + exp(alphaStem + betaStem * DVI) + 1.0; // dimensionless
131 
132  double kRoot = kcoeff(alphaRoot, betaRoot, DVI, kDenom); // dimensionless
133  double kStem = kcoeff(alphaStem, betaStem, DVI, kDenom); // dimensionless
134  double kLeaf = kcoeff(alphaLeaf, betaLeaf, DVI, kDenom); // dimensionless
135  double kGrain = 1.0 / kDenom; // dimensionless
136 
137  // Give option for rhizome to contribute to growth during the emergence stage,
138  // kRhizome_emr is an input parameter and should be non-positive.
139  // To ignore rhizome, set kRhizome_emr to 0 in input parameter file, and
140  // adjust initial leaf, stem, and root biomasses accordingly.
141  double kRhizome;
142  if (DVI < 0) {
143  kRhizome = kRhizome_emr; //dimensionless
144  } else {
145  kRhizome = 0.0; // dimensionless
146  }
147 
148  // Update the output quantities
149  update(kStem_op, kStem); //dimensionless
150  update(kLeaf_op, kLeaf); //dimensionless
151  update(kRoot_op, kRoot); //dimensionless
152  update(kRhizome_op, kRhizome); //dimensionless
153  update(kGrain_op, kGrain); //dimensionless
154 }
