13 /* Ball Berry stomatal conductance function */
14 double ball_berry(double assimilation, // mol / m^2 / s
15  double atmospheric_co2_concentration, // mol / mol
16  double atmospheric_relative_humidity, // Pa / Pa
17  double beta0, // mol / m^2 / s
18  double beta1) // dimensionless from [mol / m^2 / s] / [mol / m^2 / s]
19 {
20  const double gbw = 1.2; // mol / m^2 / s. Boundary-layer conductance. Collatz et al. (1992) Aust. J. Plant Physiol. pg. 526. The units in the manuscript, micromole / m^2 / s, are wrong . They are actually mol / m^2 / s.
21  double gswmol; // mol / m^2 / s. stomatal conductance to water vapor.
22 
23  if (assimilation > 0) {
24  const double Cs = atmospheric_co2_concentration - (1.4 / gbw) * assimilation; // mol / mol.
25  if (Cs < 0.0) {
26  throw std::range_error("Thrown in ball_berry: Cs is less than 0.");
27  }
28 
29  double acs = assimilation / Cs;
30  if (acs < 1e-6) {
31  acs = 1e-6;
32  }
33 
34  /* Calculate leaf surface relative humidity, hs, from the quadratic equation:
35  * a * hs^2 + b * hs + c = 0
36  *
37  * This can be derived as follows:
38  * At steady-state
39  * E = gs * (hs - hi) -- Equation 1
40  * and
41  * E = gb * (ha - hs) -- Equation 2
42  *
43  * Substitute gs in equation 1 using the Ball-Berry model:
44  * gs = b1 * A * hs / cs + b0
45  *
46  * Where A is assimilation rate, hs is relative humidity at the surface of the leaf, and cs is the CO2 mole fraction at the surface of the leaf.
47  *
48  * Assume hi = 1 based on saturation of water vapor in the interal airspace of a leaf.
49  * Use the equality of equations 1 and 2 to solve for hs, and it's a quadratic with the coefficients given in the code.
50  */
51  const double a = beta1 * acs; // Equivalent to a = beta1 * assimilation / cs
52  const double b = beta0 + gbw - beta1 * acs; // Equivalent to b = beta0 + gbw - beta1 * assimilation / cs
53  const double c = -atmospheric_relative_humidity * gbw - beta0;
54 
55  const double root_term = b * b - 4 * a * c;
56  const double hs = (-b + sqrt(root_term)) / (2 * a);
57 
58  gswmol = beta1 * hs * assimilation / Cs + beta0; // Ball-Berry equation (Collatz 1991, equation 1).
59  } else {
60  /* Set stomatal conductance to the minimum value, beta0 */
61  gswmol = beta0;
62  // hs = (beta0 + atmospheric_relative_humidity * gbw) / (beta0 + gbw)
63  }
64 
65  if (gswmol <= 0) {
66  gswmol = 1e-2;
67  }
68 
69  return gswmol * 1000; // mmol / m^2 / s
70 }
