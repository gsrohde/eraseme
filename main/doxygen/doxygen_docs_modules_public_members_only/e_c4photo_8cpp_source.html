96  double b1 = -((Vp - Rm + gs * Cm) + (Vcmax - Rd) +
97  gs * (Kc * (1 + Om1 / Ko1)) + ((alpha / 0.047) * (gammaStar * Vcmax + Rd * Kc / Ko1)));
98  double c1 = (Vcmax - Rd) * (Vp - Rm + gs * Cm) - (Vcmax * gs * gammaStar * Om1 + Rd * gs * Kc * (1 + Om1 / Ko1));
99

101  if (c3 < 0) {
102  c3 = 0;
103  }
104  double Ac0 = (-b1 - sqrt(c3)) / 2 * a1;
105 
106  double AcLCO2 = (Cm * Vpmax / (Cm + Kp)) - Rm + gs * Cm;
107 
108  if (Ac0 < AcLCO2) {
109  Ac = Ac0;
110  } else {
111  Ac = AcLCO2;
112  }
113 
114  if (Aj < Ac) {
115  A = Aj;
116  } else {
117  A = Ac;
118  }
119 
120  return A;
