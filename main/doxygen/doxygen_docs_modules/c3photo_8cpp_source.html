21 quantity<dimensionless> arrhenius_exponent(quantity<dimensionless> c,
22  quantity<energy_over_amount> activation_energy,
23  quantity<temperature> temperature)
24 {
25  const quantity<energy_over_temperature_amount> R = physical_constants::ideal_gas_constant * joule / kelvin / mole;
26  return exp(c - activation_energy / (R * temperature));
27 }

29 struct c3_str c3photoC(
30  double _Qp,
31  double _Tleaf,
32  double RH,
33  double _Vcmax0,
34  double _Jmax0,
35  double _TPU_rate_max,
36  double _Rd0,
37  double bb0,
38  double bb1,
39  double Gs_min,
40  double Ca,
41  double AP,
42  double _O2,
43  double thet,
44  double StomWS,
45  int water_stress_approach,
46  double electrons_per_carboxylation,
47  double electrons_per_oxygenation)
48 {
49  // Assign units to the input quantities. The parameters can be renamed and
50  // this section can be removed when call functions that call c3photoC() are
51  // also using units.
52  const quantity<flux> Rd0 = _Rd0 * 1e-6 * mole / square_meter / second;
53  const quantity<flux> Vcmax0 = _Vcmax0 * 1e-6 * mole / square_meter / second;
54  const quantity<mole_fraction> atmospheric_oxygen_mole_fraction = _O2 * 1e-3 * mole / mole;
55  const quantity<flux> Jmax0 = _Jmax0 * 1e-6 * mole / square_meter / second;
56  const quantity<temperature> leaf_temperature = (_Tleaf + conversion_constants::celsius_to_kelvin) * kelvin;
57  const quantity<flux> Qp = _Qp * 1e-6 * mole / square_meter / second;
58  const quantity<flux> Gsw_min = Gs_min * mole / square_meter / second;
59 
60  const quantity<pressure> atmospheric_pressure = AP * pascal;
61  const quantity<dimensionless> leaf_reflectance = 0.2;
62  const quantity<flux> maximum_tpu_rate = _TPU_rate_max * 1e-6 * mole / square_meter / second;
63 
64  /* Temperature corrections are from the following sources:
65  Bernacchi et al. (2003) Plant, Cell and Environment, 26(9), 1419-1430.
66  https://doi.org/10.1046/j.0016-8025.2003.01050.x
67  Bernacchi et al. (2001) Plant, Cell and Environment, 24(2), 253-259.
68  https://doi.org/10.1111/j.1365-3040.2001.00668.x */
69  /* Note: Values in Dubois and Bernacchi are incorrect. */
70  const quantity<mole_fraction> Kc = 1e-6 * arrhenius_exponent(38.05, 79.43e3 * joule / mole, leaf_temperature);
71  const quantity<mole_fraction> Ko = 1e-3 * arrhenius_exponent(20.30, 36.38e3 * joule / mole, leaf_temperature);
72  const quantity<mole_fraction> Gstar = 1e-6 * arrhenius_exponent(19.02, 37.83e3 * joule / mole, leaf_temperature);
73  const quantity<flux> Vcmax = Vcmax0 * arrhenius_exponent(26.35, 65.33e3 * joule / mole, leaf_temperature);
74  const quantity<flux> Jmax = Jmax0 * arrhenius_exponent(17.57, 43.54e3 * joule / mole, leaf_temperature);
75  const quantity<flux> Rd = Rd0 * arrhenius_exponent(18.72, 46.39e3 * joule / mole, leaf_temperature);
76 
77  const double leaf_temperature_celsius = leaf_temperature.value() - conversion_constants::celsius_to_kelvin;
78  const quantity<dimensionless> theta =
79  thet + 0.018 * leaf_temperature_celsius - 3.7e-4 * pow(leaf_temperature_celsius, 2);
80 
81  /* Light limited */
82  const quantity<dimensionless> dark_adapted_phi_PSII =
83  0.352 + 0.022 * leaf_temperature_celsius -
84  3.4 * pow(leaf_temperature_celsius, 2) / 10000; // Bernacchi et al. (2003). See reference above.
85  const quantity<flux> I2 = Qp * dark_adapted_phi_PSII * (1.0 - leaf_reflectance) / 2.0;
86 
87  const quantity<flux> J = (Jmax + I2 - root<2>(pow<2>(Jmax + I2) - 4.0 * theta * I2 * Jmax)) / (2.0 * theta);
88 
89  const quantity<mole_fraction> Oi = atmospheric_oxygen_mole_fraction * solo(leaf_temperature_celsius);
90 
91  if (Ca <= 0) {
92  Ca = 1e-4;
93  }
94 
95  const quantity<pressure> Ca_pa = Ca * 1e-6 * atmospheric_pressure; // Pa.
96 
97  quantity<pressure> Ci_pa = 0.0 * pascal;
98  quantity<flux> Vc;
99  quantity<flux> Tol = 0.01 * 1e-6 * mole / square_meter / second;
100  quantity<flux> Gs;
101  quantity<flux> co2_assimilation_rate = 0 * mole / square_meter / second;
102  quantity<mole_fraction> Ci;
103 
104  /* TPU rate temperature dependence from Fig. 7, Yang et al. (2016) Planta,
105  243, 687-698. https://doi.org/10.1007/s00425-015-2436-8 */
106  const double TPU_c = 25.5; // dimensionless. fitted constant
107  const quantity<energy_over_amount> Ha = 62.99e3 * joule / mole; // enthalpy of activation
108  const quantity<energy_over_temperature_amount> S = 0.588e3 * joule / kelvin / mole; // entropy
109  const quantity<energy_over_amount> Hd = 182.14e3 * joule / mole; // enthalpy of deactivation
110  const quantity<energy_over_temperature_amount> R = physical_constants::ideal_gas_constant * joule / kelvin / mole; // gas constant
111  const quantity<dimensionless> top = leaf_temperature.value() * arrhenius_exponent(TPU_c, Ha, leaf_temperature);
112  const quantity<dimensionless> bot = 1.0 + arrhenius_exponent(S / R, Hd, leaf_temperature);
113  double TPU_rate_scaler = top.value() / bot.value(); // dimensionless
114 
115  /* In Yang et al. the equation for `TPU_rate_scaler` is described as
116  producing the TPU rate relative to 25 degrees C. However, it does not do
117  that. Here we calculate the value of that equation at 25 degrees C, then
118  divide `TPU_rate_scaler` by that value to get a rate relative to 25
119  degrees C. */
120  const double TPU_rate_scaler25 = 306.742; // dimensionless. This is `top / bottom` at 25 degrees C.
121  TPU_rate_scaler /= TPU_rate_scaler25; // dimensionless. Normalize to 25 degrees C.
122 
123  /* The alpha constant for calculating Ap is from
124  Eq. 2.26, von Caemmerer, S. Biochemical models of leaf photosynthesis.*/
125  double alpha_TPU = 0.0; // dimensionless. Without more information, alpha=0 is often assumed.
126 
127  int iterCounter = 0;
128  int max_iter = 1000;
129  while (iterCounter < max_iter) {
130  quantity<flux> OldAssim = co2_assimilation_rate;
131 
132  /* Rubisco limited carboxylation */
133  Ci = Ci_pa / atmospheric_pressure;
134  quantity<flux> Ac1 = Vcmax * (Ci - Gstar);
135  double Ac2 = Ci + Kc * (1 + Oi / Ko);
136  quantity<flux> Ac = Ac1 / Ac2;
137 
138  /* Light limited portion */
139  quantity<flux> Aj1 = J * (Ci - Gstar);
140  double Aj2 = electrons_per_carboxylation * Ci + 2.0 * electrons_per_oxygenation * Gstar;
141  quantity<flux> Aj = Aj1 / Aj2;
142  if (Aj < 0.0 * mole / square_meter / second) {
143  Aj = 0.0 * mole / square_meter / second;
144  }
145 
146  /* Triose phosphate utilization limited */
147  quantity<flux> Ap = 3.0 * maximum_tpu_rate * (Ci - Gstar) / (Ci - (1.0 + 1.5 * alpha_TPU) * Gstar);
148  Ap = Ap * TPU_rate_scaler;
149 
150  if (Ac < Aj && Ac < Ap) {
151  Vc = Ac;
152  } else if (Aj < Ac && Aj < Ap) {
153  Vc = Aj;
154  } else if (Ap < Ac && Ap < Aj) {
155  if (Ap < 0 * mole / square_meter / second) {
156  Ap = 0 * mole / square_meter / second;
157  }
158  Vc = Ap;
159  }
160 
161  co2_assimilation_rate = Vc - Rd;
162 
163  if (water_stress_approach == 0) {
164  co2_assimilation_rate *= quantity<dimensionless>(StomWS);
165  }
166 
167  Gs = ball_berry(co2_assimilation_rate.value(), Ca * 1e-6, RH, bb0, bb1) * 1e-3 * mole / square_meter / second;
168 
169  if (water_stress_approach == 1) {
170  Gs = Gsw_min + StomWS * (Gs - Gsw_min);
171  }
172 
173  if (Gs <= 0 * mole / square_meter / second) {
174  Gs = 1e-5 * 1e-3 * mole / square_meter / second;
175  }
176 
177  Ci_pa = Ca_pa - co2_assimilation_rate * 1.6 * atmospheric_pressure / Gs;
178 
179  if (Ci_pa < 0 * pascal) {
180  Ci_pa = 1e-5 * pascal;
181  }
182 
183  if (abs(OldAssim - co2_assimilation_rate) < Tol) {
184  break;
185  }
186 
187  ++iterCounter;
188  }
189 
190  struct c3_str result;
191  result.Assim = co2_assimilation_rate.value() * 1e6; // micromole / m^2 / s.
192  result.Gs = Gs.value() * 1e3; // mmol / m^2 / s.
193  result.Ci = Ci.value() * 1e6; // micromole / mol.
194  result.GrossAssim = (co2_assimilation_rate.value() + Rd.value()) * 1e6; // micromole / m^2 / s.
195  return result;
196 }
