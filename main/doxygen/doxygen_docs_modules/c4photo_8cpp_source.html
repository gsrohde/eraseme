15 struct c4_str c4photoC(double Qp, // micromole / m^2 / s
16  double leaf_temperature, // degrees C
17  double relative_humidity, // dimensionless from Pa / Pa
18  double vmax, // micromole / m^2 / s
19  double alpha, // mol / mol
20  double kparm, // mol / m%2 / s
21  double theta,
22  double beta,
23  double Rd,
24  double bb0,
25  double bb1,
26  double Gs_min, // mmol / m^2 / s
27  double StomaWS,
28  double Ca, // micromole / mol
29  double atmospheric_pressure, // Pa
30  int water_stress_approach,
31  double upperT,
32  double lowerT)
33 {
34 
35  constexpr double k_Q10 = 2; // dimensionless. Increase in a reaction rate per temperature increase of 10 degrees Celsius.
36 
37  double Csurface = Ca * 1e-6 * atmospheric_pressure; // Pa
38  double InterCellularCO2 = Csurface * 0.4; // Pa. Use an initial guess.
39  double kT = kparm * pow(k_Q10, (leaf_temperature - 25.0) / 10.0); // dimensionless
40 
41  // Collatz 1992. Appendix B. Equation set 5B.
42  double Vtn = vmax * pow(2, (leaf_temperature - 25.0) / 10.0); // micromole / m^2 / s
43  double Vtd = (1 + exp(0.3 * (lowerT - leaf_temperature))) * (1 + exp(0.3 * (leaf_temperature - upperT))); // dimensionless
44  double VT = Vtn / Vtd; // micromole / m^2 / s
45 
46  // Collatz 1992. Appendix B. Equation set 5B.
47  double Rtn = Rd * pow(2, (leaf_temperature - 25) / 10); // micromole / m^2 / s
48  double Rtd = 1 + exp(1.3 * (leaf_temperature - 55)); // dimensionless
49  double RT = Rtn / Rtd; // micromole / m^2 / s
50 
51  // Collatz 1992. Appendix B. Equation 2B.
52  double b0 = VT * alpha * Qp;
53  double b1 = VT + alpha * Qp;
54  double b2 = theta;
55 
56  /* Calculate the 2 roots */
57  double M1 = (b1 + sqrt(b1 * b1 - 4 * b0 * b2)) / 2 / b2;
58  double M2 = (b1 - sqrt(b1 * b1 - 4 * b0 * b2)) / 2 / b2;
59 
60  double M = M1 < M2 ? M1 : M2; // Use the smallest root.
61 
62  double Assim, Gs;
63  {
64  double OldAssim = 0.0, Tol = 0.1, diff;
65  unsigned int iterCounter = 0;
66  unsigned int constexpr max_iterations = 50;
67  do {
68  // Collatz 1992. Appendix B. Equation 3B.
69  double kT_IC_P = kT * InterCellularCO2 / atmospheric_pressure * 1e6; // micromole / m^2 / s
70  double a = M * kT_IC_P;
71  double b = M + kT_IC_P;
72  double c = beta;
73 
74  double gross_assim = (b - sqrt(b * b - 4 * a * c)) / 2 / c; // micromole / m^2 / s
75 
76  Assim = gross_assim - RT; // micromole / m^2 / s.
77 
78  if (water_stress_approach == 0) Assim *= StomaWS;
79 
80  Gs = ball_berry(Assim * 1e-6, Ca * 1e-6, relative_humidity, bb0, bb1); // mmol / m^2 / s
81  if (water_stress_approach == 1) Gs = Gs_min + StomaWS * (Gs - Gs_min);
82 
83  if (iterCounter > max_iterations - 10)
84  Gs = bb0 * 1e3; // mmol / m^2 / s. If it has gone through this many iterations, the convergence is not stable. This convergence is inapproriate for high water stress conditions, so use the minimum gs to try to get a stable system.
85 
86  //Rprintf("Counter %i; Ci %f; Assim %f; Gs %f; leaf_temperature %f\n", iterCounter, InterCellularCO2 / atmospheric_pressure * 1e6, Assim, Gs, leaf_temperature);
87  InterCellularCO2 = Csurface - Assim * 1e-6 * 1.6 * atmospheric_pressure / (Gs * 0.001); // Pa
88 
89  if (InterCellularCO2 < 0)
90  InterCellularCO2 = 1e-5;
91 
92  diff = fabs(OldAssim - Assim); // micromole / m^2 / s
93 
94  OldAssim = Assim; // micromole / m^2 / s
95 
96 
97  } while (diff >= Tol && ++iterCounter < max_iterations);
98  //if (iterCounter > 49)
99  //Rprintf("Counter %i; Ci %f; Assim %f; Gs %f; leaf_temperature %f\n", iterCounter, InterCellularCO2 / atmospheric_pressure * 1e6, Assim, Gs, leaf_temperature);
100  }
101 
102  double Ci = InterCellularCO2 / atmospheric_pressure * 1e6; // micromole / mol
103 
104  struct c4_str result {
105  .Assim = Assim, // micromole / m^2 /s
106  .Gs = Gs, // mmol / m^2 / s
107  .Ci = Ci, // micromole / mol
108  .GrossAssim = Assim + RT // micromole / m^2 / s
109  };
110 
111  return result;
112 }
