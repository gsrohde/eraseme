383  // Check whether the plant is illuminated
384  // Rather than basing L(t) on specified values for dusk and dawn

387  // This function is 0 for no sunlight and saturates to 1 when

401  update(P_op, p7 * (1.0 - L) * (1.0 - P) - m11 * P * L);

404  update(LHY_prot_op, (p2 + p1 * L) * LHY_mRNA - m3 * LHY_prot - p3 * pow(LHY_prot, c) / (pow(LHY_prot, c) + pow(g3, c)));
405  update(TOC1_mRNA_op, 1.0 * n2 * g4 / (g4 + EC) * (pow(g5, e) / (pow(g5, e) + pow(LHY_prot, e))) - TOC1_mRNA * m5);
406  update(PRR9_prot_op, p8 * PRR9_mRNA - (m13 + m22 * (1.0 - L)) * PRR9_prot);
407  update(PRR5_NI_mRNA_op, 1.0 * (n10 * pow(LHY_prot_modif, e) / (pow(g12, e) + pow(LHY_prot_modif, e)) + n11 * pow(PRR7_prot, b) / (pow(g13, b) + pow(PRR7_prot, b))) - m16 * PRR5_NI_mRNA);
408  update(PRR5_NI_prot_op, p10 * PRR5_NI_mRNA - (m17 + m24 * (1.0 - L)) * PRR5_NI_prot);
409  update(GI_prot_cytoplasm_op, p11 * GI_mRNA - m19 * GI_prot_cytoplasm - p12 * L * ZTL * GI_prot_cytoplasm + p13 * GI_ZTL * (1.0 - L) - p17 * ELF3_cytoplasm * GI_prot_cytoplasm - p28 * GI_prot_cytoplasm + p29 * Gn);
410  update(TOC1_prot_op, p4 * TOC1_mRNA - m8 * TOC1_prot - (m6 + m7 * (1.0 - L)) * TOC1_prot * (p5 * ZTL + GI_ZTL));
411  update(ZTL_op, 1.0 * p14 - m20 * ZTL - p12 * L * ZTL * GI_prot_cytoplasm + p13 * GI_ZTL * (1.0 - L));
412  update(EC_op, p26 * LUX_prot * e34 - m36 * EC * COP1_nuclear_night - m37 * EC * COP1_nuclear_day - m32 * EC * (1.0 + p24 * L * pow(EGn + Gn, d) / (pow(g7, d) + pow(EGn + Gn, d))));
413  update(GI_mRNA_op, 1.0 * (q2 * L * P + pow(g15, e) / (pow(g15, e) + pow(LHY_prot, e)) * g14 / (g14 + EC) * n12) - GI_mRNA * m18);
414  update(PRR9_mRNA_op, 1.0 * q3 * L * P + g8 / (g8 + EC) * (n4 + n7 * pow(LHY_prot, e) / (pow(LHY_prot, e) + pow(g9, e))) - m12 * PRR9_mRNA);
415  update(PRR7_mRNA_op, 1.0 * (n8 * pow(LHY_prot + LHY_prot_modif, e) / (pow(g10, e) + pow(LHY_prot + LHY_prot_modif, e)) + n9 * pow(PRR9_prot, f) / (pow(g11, f) + pow(PRR9_prot, f))) - m14 * PRR7_mRNA);
416  update(PRR7_prot_op, p9 * PRR7_mRNA - PRR7_prot * (m15 + m23 * (1.0 - L)));
417  update(ELF4_mRNA_op, n13 * (pow(g6, e) / (pow(g6, e) + pow(LHY_prot, e))) * g2 / (g2 + EC) - ELF4_mRNA * m34);
418  update(ELF4_prot_op, p23 * ELF4_mRNA - m35 * ELF4_prot - p25 * ELF3_nuclear * ELF4_prot + p21 * e34);

423  update(ELF3_cytoplasm_op, p16 * ELF3_mRNA - m9 * ELF3_cytoplasm * COP1_cytoplasm - p17 * ELF3_cytoplasm * GI_prot_cytoplasm - p19 * ELF3_cytoplasm + p20 * ELF3_nuclear);
424  update(ELF3_nuclear_op, p19 * ELF3_cytoplasm - p20 * ELF3_nuclear - m29 * ELF3_nuclear * COP1_nuclear_night - m30 * ELF3_nuclear * COP1_nuclear_day - p25 * ELF3_nuclear * ELF4_prot + p21 * e34 - p17 * ELF3_nuclear * Gn);
425  update(COP1_nuclear_night_op, p6 * COP1_cytoplasm - n6 * L * P * COP1_nuclear_night - n14 * COP1_nuclear_night - m27 * COP1_nuclear_night * (1.0 + p15 * L));
