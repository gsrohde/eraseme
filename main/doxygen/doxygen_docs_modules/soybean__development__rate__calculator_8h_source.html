147 void soybean_development_rate_calculator::do_operation() const
148 {
149  double soybean_development_rate; // day^-1
150 
151  if (time < sowing_time) {
152  // The seeds haven't been sown yet, so no development should occur
153  soybean_development_rate = 0; // day^-1
154  } else if (DVI < -1) {
155  // error, DVI out of bounds, this should never occur unless initial DVI
156  // state is less than -1.
157  soybean_development_rate = 0; // day^-1
158 
159  } else if (DVI < 0) {
160  // 1. Sowing to emergence
161 
162  double temp_diff = temp - Tbase_emr; // degrees C
163  soybean_development_rate = std::max(0.0, temp_diff / TTemr_threshold); // day^-1
164 
165  } else if (DVI < 0.333) {
166  // 2a. Emergence - V0 (cotyledon stage); r = Rmax * f(T)
167 
168  soybean_development_rate = Rmax_emrV0 * tempFunc(temp, Tmin_emrV0, Topt_emrV0, Tmax_emrV0) / 3.0; // day^-1
169  } else if (DVI < 0.667) {
170  // 2b. V0 (cotyledon) - R0 (end of floral induction); r = Rmax * f(P)
171 
172  const double Rmax_V0R0 = 0.0294 + 0.1561 / maturity_group; // day^-1;
173  // Setiyono et al., 2007, Fig. 12a
174 
175  const double Popt_V0R0 = 12.759 - 0.388 * maturity_group -
176  0.058 * pow(maturity_group, 2); // hours; Setiyono et al., 2007, Fig 4
177  const double Pcrit_V0R0 = 27.275 - 0.493 * maturity_group -
178  0.066 * pow(maturity_group, 2); // hours; Setiyono et al., 2007, Fig 4
179 
180  double day_length_civil = 1.072 * day_length; // hours;
181  // Converts day length calculated based on solar elevation = -0.83 degrees
182  // to civil day length (based on solar elevation angle = -6 degrees)
183 
184  soybean_development_rate = Rmax_V0R0 *
185  photoFunc(day_length_civil, Popt_V0R0, Pcrit_V0R0) / 3.0; // day^-1
186 
187  } else if (DVI < 1) {
188  // 2c. R0 (end of floral induction) - R1 (flowering); r = Rmax * f(T)
189 
190  const double Rmax_R0R1 = (0.2551 + 0.0965 * maturity_group) /
191  (1 + 2.1107 * maturity_group); // day^-1; Setiyono et al., 2007, Fig 11a
192 
193  soybean_development_rate = Rmax_R0R1 *
194  tempFunc(temp, Tmin_R0R1, Topt_R0R1, Tmax_R0R1) / 3.0; // day^-1
195 
196  } else {
197  // 3. Reproductive stages, R1 (flowering) - R7 (maturity);
198  // r = Rmax * f(T) * f(P)
199 
200  const double Rmax_R1R7 = (0.0563 + 0.0228 * maturity_group) /
201  (1 + 1.9683 * maturity_group); // day^-1; Setiyono et al., 2007, Fig 11d
202 
203  const double Popt_R1R7 = 10.6595 + 2.9706 / maturity_group; // hrs;
204  // Setiyono et al., 2007, Fig 12c
205 
206  const double Pcrit_R1R7 = 16.1257 + 4.3143 / maturity_group; // hrs;
207  // Setiyono et al., 2007, Fig 12c
208 
209  soybean_development_rate = Rmax_R1R7 *
210  tempFunc(temp, Tmin_R1R7, Topt_R1R7, Tmax_R1R7) *
211  photoFunc(day_length, Popt_R1R7, Pcrit_R1R7); // day^-1
212  }
213 
214  double development_rate_per_hour = soybean_development_rate / 24.0; // hr^-1
215 
216  // Update the output quantity list
217  update(development_rate_per_hour_op, development_rate_per_hour);
218 }

220 double tempFunc(double T, double Tmin, double Topt, double Tmax)
221 {
222  double fT; // dimensionless
223 
224  if (T > Tmin && T < Tmax) {
225  double alpha = log(2.0) / log((Tmax - Tmin) / (Topt - Tmin)); // dimensionless
226  double fT_num_pt1 = 2.0 * pow(T - Tmin, alpha) * pow(Topt - Tmin, alpha); // dimensionless
227  double fT_num_pt2 = pow(T - Tmin, 2.0 * alpha); // dimensionless
228  double fT_denom = pow(Topt - Tmin, 2.0 * alpha); // dimensionless
229 
230  fT = (fT_num_pt1 - fT_num_pt2) / fT_denom; // dimensionless
231  } else {
232  fT = 0.0; // dimensionless
233  }
234 
235  return fT; // dimensionless
236 }

238 double photoFunc(double P, double Popt, double Pcrit)
239 {
240  double m = 3.0; // hrs; Setiyono et al., 2007 Equation 4, value in text below equation
241  double fP; // dimensionless
242 
243  if (P <= Popt) {
244  fP = 1.0; // dimensionless
245 
246  } else if (P < Pcrit) {
247  double alpha = log(2.0) / log(1.0 + (Pcrit - Popt) / m); // dimensionless
248  double fP_pt1 = 1.0 + (P - Popt) / m; // dimensionless
249  double fP_pt2 = (Pcrit - P) / (Pcrit - Popt); // dimensionless
250  double fP_pt3 = (Pcrit - Popt) / m; // dimensionless
251 
252  fP = pow(fP_pt1 * pow(fP_pt2, fP_pt3), alpha); // dimensionless
253 
254  } else {
255  fP = 0.0; // dimensionless
256  }
257 
258  return fP; // dimensionless
259 }
