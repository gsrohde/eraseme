120 void dynamical_system::update_drivers(double time_indx)
121 {
122  // Find two closest surrounding integers:
123  int t1 = std::floor(time_indx);
124  int t2 = t1 + 1; // note t2 - t1 = 1
125  for (const auto& x : driver_quantity_ptr_pairs) {
126  // Use linear interpolation to find value at time_indx:
127  auto value_at_t1 = (*(x.second))[t1];
128  auto value_at_t2 = (*(x.second))[t2];
129  auto value_at_time_indx =
130  value_at_t1 + (time_indx - t1) * (value_at_t2 - value_at_t1);
131 
132  *(x.first) = value_at_time_indx;
133  }
134 }
